sr = [0.01, 0.04, 0.1, 0.25, 0.5]
x = [i + 1 for i in range(len(sr))]
markers = [".", "v", "p", "*", "d","x", "s", 'o']
colors = ['tab:cyan', 'tab:pink', 'tab:orange', None,None, 'green','blue', 'red' ]

for i, method in enumerate(methods):
    psnr = [
        get_psnr_by_sr(
            method, _sr, datasets=["Set5", "Set14", "Set11", "BSD100", "Urban100"]
        )
        for _sr in sr
    ]
    plt.plot(x, psnr, marker=markers[i], color=colors[i])
    print(psnr)

# 设置图片的右边框和上边框为不显示

ax.spines["top"].set_color("none")
